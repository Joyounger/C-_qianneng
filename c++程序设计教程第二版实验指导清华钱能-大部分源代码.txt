//=============================================================================================
//Function Definition
double C(int a,int b)
{
	double r(1),d(b);
	for(int i(1);i<=b;i++)
	{
		r*=a;
		a--;
		r/=d;
		d--;
	}
	return r;
}

{

	for(int i(0);i<(t-2);i++)
	{
		if(f==false)

int ReadBit(int number,unsigned int* p)
{
	if(p[number/32] & (1<<(number%32)))
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

void SetBit(int number,unsigned int* p)
{
	p[number/32] &= ~(1<<(number%32));
}

void factor(unsigned int* p)
{
	for(int i=2;i<=10000;i++)
	{
		if(p[i/32]&(1<<(i%32)))
		{
			for(int j=i*i;j<100000000;j += i)
			{
				p[j/32] &= ~(1<<(j%32));
			}
		}
	}
}

v_cs encipher(v pass,v text)
{
	v_cs vcs_s;
	getpass(pass,true);
	for(int i(0);i<=text.size();i++)
	{
		int passnumber=getpass(pass,false);
		if((text[i]+passnumber==0))
		{
			vcs_s.push_back(' ');
		}
		else
		{
			if((text[i]+passnumber==90)||(text[i]+passnumber==180))
			{
				vcs_s.push_back('z');
			}
			else
			{
				vcs_s.push_back((char)((text[i]+passnumber)%90+32));
			}
		}
	}
	return vcs_s;
}

v_cs decod(v pass,v text)
{
	v_cs vcs_s;
	getpass(pass,true);
	for(int i(0);i<=(text.size()-1);i++)
	{
		int passnumber=getpass(pass,false);
		if((text[i]-passnumber<0))
		{
			vcs_s.push_back((char)(text[i]-passnumber+90+32));
		}
		else
		{
			vcs_s.push_back((char)(text[i]-passnumber+32));
		}
	}
	return vcs_s;
}

int getpass(v pass,bool flg)
{
	static int c(0);
	if(flg)
	{
		c=0;
		return 0;
	}
	else
	{
		if(c/pass.size()==0)
		{
			c++;
			return (pass[c-1]);
		}
		else
		{
			c++;
			return(pass[(c-1)-(((c-1)/pass.size())*pass.size())]);
		}
	}
}


void Print(v_cs text)
{
	for(int i(0);i<=(text.size()-1);i++)
	{
		cout<<text[i];
	}
	cout<<endl;

//Function Definition
int ReadBit(int number,unsigned int* p)
{
	if(p[number/32] & (1<<(number%32)))
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

void SetBit(int number,unsigned int* p)
{
	p[number/32] &= ~(1<<(number%32));
}

int Yang(int a,int b)
{
	if((a==b)||(a==1)||(b==1))
	{
		return 1;
	}
	else
	{
		return Yang(a-1,b-1)+Yang(a-1,b);
	}
}
